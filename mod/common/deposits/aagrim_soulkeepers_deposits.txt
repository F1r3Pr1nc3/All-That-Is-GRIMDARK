d_aa_exodite_enclave = {
	time = 360

	is_for_colonizable = yes
	category = deposit_cat_rare_blocker

	potential = { always = no }

	on_cleared = {
		if = {
			limit = { exists = owner }
			if = {
				limit = { owner = { has_soulkeepers_origin = yes } }
				create_pop_group = { species = owner }
				random = { chance = 50 create_pop_group = { species = owner } }
			}
			else = {
				owner = { add_resource = { minerals = 250 } }
			}
		}
	}

	resources = {
		category = deposit_blockers
		cost = {
			trigger = {
				owner = { has_soulkeepers_origin = no }
			}
			energy = 1000
		}
		cost = {
			trigger = {
				owner = { has_soulkeepers_origin = yes }
			}
			influence = 100
		}
		produces = {
			trigger = {
				owner = { has_soulkeepers_origin = yes }
			}
			food = 5
			minerals = 3
			society_research = 3
		}
	}

	planet_modifier = { planet_max_districts_add = -1 planet_max_buildings_add = 1 }

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_soulkeepers_origin = yes }
		}
		modifier = { planet_amenities_add = 200 job_exodite_add = 100 }
	}

	ai_weight = { weight = 0 }
}
