cultist = {
	category = worker
	swappable_data = {
		default = {
			condition_string = WORKER_JOB_TRIGGER
			building_icon = building_luxury_residence
		}
	}

	is_capped_by_modifier = no
	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
	}

	possible = {
		can_take_servant_job = no
		hidden_trigger = { exists = owner }
		owner = {
			OR = {
				has_origin = origin_shroudsinners
				has_origin = origin_shroudmutated
				has_origin = origin_paindrinkers
			}
		}
		OR = {
			is_unemployed = yes
		}
	}

	resources = {
		category = planet_culture_workers
		produces = {
			unity = 1
		}
	}

	planet_modifier = {
		planet_amenities_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			any_owned_species = {
				has_trait = trait_robot_domestic_protocols
				has_trait = trait_charismatic
			}
		}
		planet_amenities_add = 0.25
	}

	triggered_planet_modifier = {
		potential = {
			any_owned_species = {
				has_trait = trait_repugnant
			}
		}
		planet_amenities_add = -0.25
	}

	weight = {
		weight = @special_job_weight
	}
}

conscript = {
	category = worker
	swappable_data = {
		default = {
			condition_string = WORKER_JOB_TRIGGER
			building_icon = building_stronghold
		}
	}

	is_capped_by_modifier = no
	possible_pre_triggers = {
		has_owner = yes
		is_being_assimilated = no
		is_being_purged = no
	}

	possible = {
		# can_take_servant_job = no
		# is_enslaved = no
		hidden_trigger = { exists = owner }
		owner = {
			OR = {
				has_origin = origin_shroudrage
				has_origin = origin_barbarichorde
			}
		}
		OR = {
			is_unemployed = yes
		}
	}

	country_modifier = {
		country_naval_cap_add = 2
	}

	triggered_country_modifier = {
		potential = {
			exists = planet
			planet = {
				has_owner = yes
				owner = { has_technology = "tech_ground_defense_planning" }
			}
		}
		country_naval_cap_add = 1
	}

	resources = {
		category = planet_soldiers
		produces = {
			trigger = {
				exists = owner
				owner = { has_valid_civic = civic_citizen_service }
			}
			unity = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			has_modifier = martial_law
		}
		planet_stability_add = 2
	}

	weight = {
		weight = @special_job_weight
	}
}
