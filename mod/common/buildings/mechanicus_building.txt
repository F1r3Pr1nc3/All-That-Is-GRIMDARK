		is_occupied_flag = no

		job_coder_per_pop = 0.05
		pop_ethic_spiritualist_attraction_mult = 0.5
	}

			minerals = 500
		}

			alloys = 6
		}

			energy = 2

		}
	}

			factor = 3

		}

			factor = 6
			branch_office_owner = { has_ethic = ethic_fanatic_spiritualist }
		}
	}
