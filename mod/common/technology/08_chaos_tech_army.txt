# tech_army_assimilation = { # icon = tech_plague_zombies
tech_plague_zombies = {
	cost = @tier2cost3
	area = society
	tier = 2
	is_rare = yes
	category = { biology } # chaos
	prerequisites = { "tech_assault_armies" } # tech_chaos_marines
	weight = @tier2weight3
	potential = {
		OR = {
			AND = {
				has_tradition = tr_genetics_resequencing # Former tech_genetic_resequencing
				is_hive_empire = yes
				can_reanimate_armies = no
			}
			has_origin = origin_shroudsick
			AND = { has_country_flag = nurgle has_country_flag = chaos_empire }
		}
	}
	modifier = { army_morale_damage_mult = 1 }
	# weight_modifier = {
	# 	modifier = { factor = 1.25 has_trait_in_council = { TRAIT = leader_trait_expertise_chaos } }
	# }
	# ai_weight = {
	# 	modifier = { factor = 1.25 has_trait_in_council = { TRAIT = leader_trait_expertise_chaos } }
	# }
	ai_weight = {
		modifier = { factor = 1.25 has_trait_in_council = { TRAIT = leader_trait_expertise_biology } }
		modifier = { factor = 1.25 has_trait_in_council = { TRAIT = leader_trait_society_focus } }
		modifier = { factor = 1.25 owner = { has_valid_civic = civic_hive_devouring_swarm } }
		modifier = { factor = 0 owner = { has_valid_civic = civic_hive_reanimated_armies } }
	}
}
