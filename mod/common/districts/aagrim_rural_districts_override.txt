district_generator = {
	expansion_planner = yes

	base_buildtime = 240

	min_for_deposits_on_planet = 3
	max_for_deposits_on_planet = 15

	show_on_uncolonized = {
		uses_district_set = standard
	}

	potential = {
		uses_district_set = standard
	}

	allow = {
		NAND = {
			exists = owner
			owner = { is_ai = yes }
			owner = { should_ai_focus_on_trade = yes }
		}
		NOT = { has_modifier = resort_colony }
	}

	destroy_trigger = {
		exists = owner
		owner = {
			is_ai = yes
			should_ai_focus_on_trade = yes
		}
	}

	conversion_ratio = 1
	convert_to = {
		district_generator_uncapped
	}

	resources = {
		category = planet_districts_generator
		cost = {
			minerals = @base_cost
		}
		upkeep = {
			energy = 1
		}
	}

	planet_modifier = {
		planet_housing_add = 2
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
			owner = { NOT = { has_origin = origin_barbarichorde } }
		}
		modifier = {
			job_technician_drone_add = 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
			owner = { NOT = { has_origin = origin_barbarichorde } }
		}
		modifier = {
			job_technician_drone_add = 2
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { NOT = { has_origin = origin_barbarichorde } }
			owner = {
				is_gestalt = no
				is_fallen_empire_spiritualist = no
			}
		}
		modifier = {
			job_technician_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_origin = origin_barbarichorde }
		}
		modifier = {
			planet_housing_add = 1
			job_burnaboy_add = 2
			job_teefdealer_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }
			owner = { NOT = { has_origin = origin_barbarichorde } }
		}
		modifier = {
			job_fe_acolyte_generator_add = 2
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
			owner = { NOT = { has_origin = origin_barbarichorde } }
		}
		text = job_technician_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner

			owner = {
				is_gestalt = no

			}
