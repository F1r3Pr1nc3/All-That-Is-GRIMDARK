@base_cost = 300
@city_cost = 500

district_city = {
	base_buildtime = 480
	is_capped_by_modifier = no
	exempt_from_ai_planet_specialization = yes

	show_on_uncolonized = {
		exists = from
		from = { is_regular_empire = yes }
		OR = {
			uses_district_set = standard
			uses_district_set = hive_world
			uses_district_set = machine_world
			uses_district_set = shattered_ring_world
		}
		NOT = { has_modifier = resort_colony }
	}

	potential = {
		OR = {
			NOT = { exists = owner } # This condition is needed for galaxy generation
			owner = { is_regular_empire = yes }
			owner = { is_primitive = yes is_hive_empire = no }
		}
		NOT = { has_deposit = d_crashed_slaver_ship }
		OR = {
			uses_district_set = district_resort
			uses_district_set = standard
			uses_district_set = hive_world
			uses_district_set = machine_world
			uses_district_set = shattered_ring_world
		}
		NOT = { has_modifier = resort_colony }
	}

	allow = {
		NOR = {
			# has_modifier = resort_colony
			has_modifier = slave_colony
		}
		# Ariphaos Patch (3.9)
		NAND = {
			exists = owner
			owner = { is_ai = yes }
			num_districts = { type = district_city value >= 3 }
			AND = {
				is_capital = yes
				num_districts = { type = district_city value >= 6 }
			}
		}
	}

	conversion_ratio = 1
	convert_to = {
		district_arcology_housing
		district_rw_city
		district_nexus
		district_hive
	}

	resources = {
		category = planet_districts_cities
		cost = {
			minerals = @city_cost
		}
		upkeep = {
			energy = 2
		}
	}

	planet_modifier = {
		planet_housing_add = 5
		planet_max_buildings_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = -1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {  }
			owner = {
				OR = {
					AND = {
						is_regular_empire = yes
						NOT = { has_origin = origin_barbarichorde }
					}
					has_origin = origin_fear_of_the_dark
				}
			}
		}
		modifier = {
			job_clerk_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_active_tradition = tr_virtuality_4 # tr_prosperity_interstellar_franchising
				NOT = { has_origin = origin_barbarichorde }
			}
		}
		modifier = {
			job_clerk_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_origin = origin_barbarichorde }
		}
		modifier = {
			job_weirdboy_add = 2
			job_bignob_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_active_tradition = tr_prosperity_public_works }
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_1
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_2
				NOT = { has_civic = civic_agrarian_idyll }
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { NOT = { has_origin = origin_barbarichorde } }
			owner = { is_regular_empire = yes }
		}
		text = job_clerk_effect_desc
	}
}
