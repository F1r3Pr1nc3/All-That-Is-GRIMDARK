district_arcology_manufactorum = {
	# icon = district_mining
	base_buildtime = 600
	zone_slots = {
		slot_city_government
		slot_city_01
		slot_city_02
	}

	show_on_uncolonized = {
		uses_district_set = imperiumhive_world
		exists = from
		from = {
			OR = {
				has_origin = origin_godruler
				has_machine_cult = yes
			}
		}
	}

	potential = {
		uses_district_set = imperiumhive_world
		exists = owner
		owner = {
			OR = {
				has_origin = origin_godruler
				has_machine_cult = yes
			}
		}
	}

	conversion_ratio = 1
	convert_to = {
		district_city
		district_nexus
		district_hive
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
		}
		produces = {
			consumer_goods = 5
			alloys = 5
		}
		upkeep = {
			energy = 3
		}
	}

	planet_modifier = {
		planet_housing_add = 1000
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		job_fabricator_add = 200
		job_artisan_drone_add = 200
		job_mining_drone_add = 600
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		job_alloy_drone_add = 200
		job_artisan_drone_add = 200
		job_mining_drone_add = 600
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		job_imperium_labourer_add = 500
		job_miner_add = 500
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 500
			owner = {
				has_monthly_income = {
					resource = minerals
					value > 100
				}
			}
		}
	}
}
