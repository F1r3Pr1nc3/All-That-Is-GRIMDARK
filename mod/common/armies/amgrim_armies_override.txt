# Imperial Legions Override for the God Ruler
imperial_legion = {
	damage = 2.5
	health = 3.0
	morale = 3.0
	morale_damage = 1.0
	collateral_damage = 0.75
	war_exhaustion = 3.0
	time = 150
	icon = GFX_army_type_imperial
	has_species = no

	resources = {
		category = armies
		cost = { minerals = 300 }

		upkeep = { energy = 3 }
	}

	potential_country = {
		OR = {
			AND = {
				is_galactic_emperor = yes
				is_active_resolution = "resolution_emperor_imperial_legions"
			}
			AND = {
				has_origin = origin_godruler
				# TODO?
			}
		}
	}

	potential = {
		OR = {
			planet = { is_capital = yes }
			owner = { is_galactic_emperor = yes }
		}
	}

	allow = {
		custom_tooltip = {
			text = imperial_legion_limit_tooltip
			owner = {
				check_variable = { which = "imperial_legion_soldier_count" value < 12 }
			}
		}
	}

	on_queued = {
		owner = {
			change_variable = { which = "imperial_legion_soldier_count" value = 1 }
		}
	}

	on_unqueued = {
		owner = {
			change_variable = { which = "imperial_legion_soldier_count" value = -1 }
		}
	}

	ai_weight = {
		base = 500
		modifier = {
			set = 0
			check_variable = { which = "imperial_legion_soldier_count" value >= 12 }
		}
	}
}
